class Main {
    public static void main(String[] args) {
    int arr[]={1,1,0,1,1,0,0,1,0,1,1,1,0};
        int prev=0;
        int count=0;
        
        for( int i=1;i<arr.length;i++){
            if(arr[i]==1){
                count++;
            }
            else {
            if(count>prev){
                
                prev=count;
                
            }
                count=0;
            }
         }
       System.out.println(prev);
  }
}
